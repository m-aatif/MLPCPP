214 bool aRectangle::intersect(aPoint pt1, aPoint pt2, bool considerTouch)
215 {
216 double dx = pt2.x() - pt1.x();
217 double dy = pt2.y() - pt1.y();
218 double sbl = dy * (left() - pt1.x()) - dx * (bottom() - pt1.y());
219 double sbr = dy * (right() - pt1.x()) - dx * (bottom() - pt1.y());
220 double stl = dy * (left() - pt1.x()) - dx * (top() - pt1.y());
221 double str = dy * (right() - pt1.x()) - dx * (top() - pt1.y());
222 bool bCheck = isPositive(sbl);
223 if (bCheck == isPositive(sbr))
224 if (bCheck == isPositive(stl))
225 if (bCheck == isPositive(str))
226 return false;
227 double xMin = min(pt1.x(), pt2.x());
228 double xMax = max(pt1.x(), pt2.x());
229 double yMin = min(pt1.y(), pt2.y());
230 double yMax = max(pt1.y(), pt2.y());
231 double xrMin = min(left(), right());
232 double xrMax = max(left(), right());
233 double yrMin = min(bottom(), top());
234 double yrMax = max(bottom(), top());
235 bool bCheck2(
236 ((xrMin <= xMin) && (xrMax >= xMin)) ||
237 ((xrMin <= xMax) && (xrMax >= xMax)) ||
238 ((xMin <= xrMin) && (xMax >= xrMin)) ||
239 ((xMin <= xrMax) && (xMax >= xrMax))
240 );
241 bool bCheck3(
242 ((yrMin <= yMin) && (yrMax >= yMin)) ||
243 ((yrMin <= yMax) && (yrMax >= yMax)) ||
244 ((yMin <= yrMin) && (yMax >= yrMin)) ||
245 ((yMin <= yrMax) && (yMax >= yrMax))
246 );
247 return (bCheck2 | bCheck3);
248 }

637 int GeometryTools::orientation(aPoint p, aPoint q, aPoint r)
638 {
639 double val = (q.y() - p.y()) * (r.x() - q.x()) - (q.x() - p.x()) * (r.y() - q.y());
640 if (val > 0)
641 // Clockwise orientation
642 return 1;
643 else if (val < 0)
644 // Counterclockwise orientation
645 return 2;
646 else
647 // Colinear orientation
648 return 0;
649 }

657 bool GeometryTools::intersectSegment(aPoint p1, aPoint q1, aPoint p2, aPoint q2)
658 {
659 // Find the 4 orientations required for
660 // the general and special cases
661 int o1 = orientation(p1, q1, p2);
662 int o2 = orientation(p1, q1, q2);
663 int o3 = orientation(p2, q2, p1);
664 int o4 = orientation(p2, q2, q1);
665
666 // General case
667 if ((o1 != o2) && (o3 != o4))
668 return true;
669
670 // Special Cases
671
672 // p1, q1 and p2 are colinear and p2 lies on segment p1q1
673 if ((o1 == 0) && onSegment(p1, p2, q1))
674 return true;
675
676 // p1, q1 and q2 are colinear and q2 lies on segment p1q1
677 if ((o2 == 0) && onSegment(p1, q2, q1))
678 return true;
679
680 // p2, q2 and p1 are colinear and p1 lies on segment p2q2
681 if ((o3 == 0) && onSegment(p2, p1, q2))
682 return true;
683
684 // p2, q2 and q1 are colinear and q1 lies on segment p2q2
685 if ((o4 == 0) && onSegment(p2, q1, q2))
686 return true;
687
688 // If none of the cases
689 return false;
690 }

692 bool GeometryTools::belongsToLine(aPoint pt0, aPoint pt, aPoint pt1)
693 {
694 double x0, y0, m, y, dx;
695 if (pt0.x() == pt1.x())
696 {
697 return (pt.x() == pt0.x());
698 }
699 else
700 {
701 x0 = pt0.x();
702 y0 = pt0.y();
703 dx = pt1.x() - x0;
704 m = (pt1.y() - y0) / dx;
705
706 y = y0 + (pt.x() - x0)*m;
707 return (y == pt.y());
708 }

710 return false;
