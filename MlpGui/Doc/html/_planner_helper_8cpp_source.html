105 {
106 reversePositionInPlace(context.navigationArea.area.at(np));

118 }

124 endian_reverse_inplace(context.vehicles.at(nv).stateVector.lastUpdate);
125 reverseOrientationInPlace(context.vehicles.at(nv).stateVector.orientation);
126 endian_reverse_inplace(context.vehicles.at(nv).stateVector.gimbalPitch);
127 endian_reverse_inplace(context.vehicles.at(nv).stateVector.linearSpeed);
128 }

130 {

132 endian_reverse_inplace(context.tasks.at(nt).altitude);

144 for (uint32_t p = 0; p < context.commands[nc].params.size(); p++)

508 double PlannerHelper::haversine(aPoint pt1, aPoint pt2)
509 {
510 double deltaLonRads = (pt2.x() - pt1.x()) * DEG2RAD;
511 double deltaLatRads = (pt2.y() - pt1.y()) * DEG2RAD;
512
513 double s1 = sin(deltaLatRads / 2);
514 double s2 = sin(deltaLonRads / 2);
515 double a = s1 * s1 + cos(pt1.y()*DEG2RAD) * cos(pt2.y()*DEG2RAD) * s2*s2;
516 return EARTH_MEAN_RADIUS * 2.0 * asin(sqrt(a));
517 }

536 void PlannerHelper::matrixRowLimits(matrix<double> mtx, int row, double &minVal, double &maxVal)
537 {
538 double vMin = 1E12, vMax = -1E12;
539
540 for (size_t c = 0; c < mtx.size2(); c++)
541 {
542 if (mtx(row, c) < vMin)
543 vMin = mtx(row, c);
544 if (mtx(row, c) > vMax)
545 vMax = mtx(row, c);
546 }
547
548 minVal = vMin;
549 maxVal = vMax;
550 }

552 void PlannerHelper::preprocessingForOptimalPath(Mission &context)
553 {
554 Position p;
555 Region r;
556 Vehicle v;
557 Task t;
558
562 //----------------------------------------------
563 p = context.navigationArea.area.at(0);
564 m_mapStuff.absMinLon = m_mapStuff.absMaxLon = p.longitude;
565 m_mapStuff.absMinLat = m_mapStuff.absMaxLat = p.latitude;
566 for (uint32_t np = 1; np < context.navigationArea.area.size(); np++)
567 {
568 p = context.navigationArea.area.at(np);
569 if (p.longitude < m_mapStuff.absMinLon)
570 m_mapStuff.absMinLon = p.longitude;
571 if (p.longitude > m_mapStuff.absMaxLon)
572 m_mapStuff.absMaxLon = p.longitude;
573 if (p.latitude < m_mapStuff.absMinLat)
574 m_mapStuff.absMinLat = p.latitude;
575 if (p.latitude > m_mapStuff.absMaxLat)
576 m_mapStuff.absMaxLat = p.latitude;
577 }
578 Position p1 = context.navigationArea.area.at(0);
579 Position p2 = context.navigationArea.area.at(2);
580 m_mapStuff.navArea = aRectangle(
581 aPoint(p1.longitude, p1.latitude),
582 aPoint(p2.longitude, p2.latitude)
583 );
584
585 if (context.forbiddenArea.size() > 0)
586 {
587 for (uint32_t na = 0; na < context.forbiddenArea.size(); na++)
588 {
589 r = context.forbiddenArea.at(na);
590 for (uint32_t np = 0; np < r.area.size(); np++)
591 {
592 p = r.area.at(np);
593 if (p.longitude < m_mapStuff.absMinLon)
594 m_mapStuff.absMinLon = p.longitude;
595 if (p.longitude > m_mapStuff.absMaxLon)
596 m_mapStuff.absMaxLon = p.longitude;
597 if (p.latitude < m_mapStuff.absMinLat)
598 m_mapStuff.absMinLat = p.latitude;
599 if (p.latitude > m_mapStuff.absMaxLat)
600 m_mapStuff.absMaxLat = p.latitude;
601 }
602 }
603 }
604
605 for (uint32_t nv = 0; nv < context.vehicles.size(); nv++)
606 {
607 v = context.vehicles.at(nv);
608 p = v.stateVector.position;
609 if (p.longitude < m_mapStuff.absMinLon)
610 m_mapStuff.absMinLon = p.longitude;
611 if (p.longitude > m_mapStuff.absMaxLon)
612 m_mapStuff.absMaxLon = p.longitude;
613 if (p.latitude < m_mapStuff.absMinLat)
614 m_mapStuff.absMinLat = p.latitude;
615 if (p.latitude > m_mapStuff.absMaxLat)
616 m_mapStuff.absMaxLat = p.latitude;
617 }
618
619 for (uint32_t nt = 0; nt < context.tasks.size(); nt++)
620 {
621 t = context.tasks.at(nt);
622 for (uint32_t np = 0; np < t.area.area.size(); np++)
623 {
624 p = context.navigationArea.area.at(np);
625 if (p.longitude < m_mapStuff.absMinLon)

638 m_mapStuff.n_distance = normalize(
639 hypot(m_mapStuff.absDeltaLon, m_mapStuff.absDeltaLat),

641 m_mapStuff.absMaxLon);
642
643 m_mapStuff.obstacleMinimumSize = 0;
644 if (context.forbiddenArea.size() > 0)
645 {
646 m_mapStuff.obstacleMinimumSize = 1.0E6;

879 void PlannerHelper::survey(Mission &context, uint32_t taskIndex, bool bLastTask)
880 {
881 pointVec path;
882 uint32_t vehicleIndex = findVehicleIndexById(context, m_mapStuff.sortedTasks[taskIndex].assignedVehicleId);
883
884 preprocessingPathPlanning(context, taskIndex, vehicleIndex);
885
886 checkOnFirstTask(0 == taskIndex, context, vehicleIndex, path);
887
888 aPoint startPoint = positionToAPoint(context.vehicles[vehicleIndex].stateVector.position);
889
890 TaskAlgorithm ta, tca;
891 int32_t algo = 0;
892 int32_t connectionAlgo = 0;
893 int32_t decompositionType = 0;
894
895 if (getTaskAlgorithmDescr("SURVEY", ta))
896 {
897 algo = getTaskAlgorithm(ta);
898 decompositionType = (int32_t)ta.parameters["Decomposition Type"];
899 }
900 switch (algo)
901 {
902 default:
903 case 2:
904 {
905 if (getTaskAlgorithmDescr("SURVEY_TRANSIT", tca))
906 {
907 connectionAlgo = getTaskAlgorithm(tca);
908 }
909
910 double cameraSpanAngle = 90.0;
911 double bigBase = haversine(m_mapStuff.start, m_mapStuff.goal);
912 double height = context.vehicles[vehicleIndex].stateVector.position.altitude;
913 double span = 2.0*height*tan(cameraSpanAngle*DEG2RAD/2.0);
914
915 double resolution = (span / EARTH_MEAN_RADIUS) * RAD2DEG;
916 double resolution2 = hypot(m_mapStuff.navArea.width(), m_mapStuff.navArea.height()) / 100;
917 double resolution3 = m_mapStuff.obstacleMinimumSize / 2;
918
919#ifdef _WINDOWS
920 TRACE(_T("TASK AREA DIAGONAL (%f) m\n"), bigBase);
921 TRACE(_T("VEHICLE ALTITUDE (%f) m\n"), height);
922 TRACE(_T("VEHICLE GROUND SPAN(%f) m\n"), span);
923 TRACE(_T("RESOLUTION (SPAN) (%f) m\n"), resolution);
924 TRACE(_T("RESOLUTION (NAV) (%f) m\n"), resolution2);
925 TRACE(_T("RESOLUTION (OBS) (%f) m\n"), resolution3);
926#endif // _WINDOWS
927
928 CoveragePlanner planner(m_mapStuff.start, m_mapStuff.goal, m_mapStuff.obstacles_rect, m_mapStuff.randArea, startPoint, resolution3);
929 planner.setDecompositionType(decompositionType);
930
931 double resolutionDivider, extraParam1 = 0.0;
932 switch (connectionAlgo)
933 {
934 case 0:
935 resolutionDivider = ta.parameters["Resolution Divider"];
936 extraParam1 = ta.parameters["Heuristic Weight"];
937
938 if (0 == resolutionDivider)
939 resolutionDivider = 50;
940 if (0 == extraParam1)
941 extraParam1 = DEFAULT_ASTAR_HEURISTIC_WEIGHT;
942
943 resolution = hypot(m_mapStuff.navArea.width(), m_mapStuff.navArea.height()) / resolutionDivider;
944 break;
945 case 3:
946 resolutionDivider = ta.parameters["Resolution Divider"];
947 if (0 == resolutionDivider)
948 resolutionDivider = 50;
949 double dx = m_mapStuff.sortedTasks[taskIndex].area.area[1].longitude - m_mapStuff.sortedTasks[taskIndex].area.area[0].longitude;
950 double dy = m_mapStuff.sortedTasks[taskIndex].area.area[1].latitude - m_mapStuff.sortedTasks[taskIndex].area.area[0].latitude;
951 resolution = hypot(dx, dy) / resolutionDivider;
952 break;
953 }
954
955 planner.setConnectionAlgorithm(connectionAlgo, resolution, extraParam1);
956
957 path = planner.plan();
958 break;
959 }
960 }
961
962 checkOnLastTask(bLastTask, context, vehicleIndex, path);
963
964 logPath(m_mapStuff.sortedTasks[taskIndex], path);
965#ifdef _WINDOWS
966 postPath(taskIndex, path);
967#endif // _WINDOWS
968
969 fillCommands(context, path, taskIndex, vehicleIndex, false);
970 }

1142 void PlannerHelper::fillCommandsForTransitFlyingVehicles(Mission &context, std::vector<aPoint> path, Task &task, uint32_t vehicleIndex, bool bLastTask)
1143 {
1144 aPoint previousWaypoint;
1145 Command cmd;
1146 double v, vMax, vAllowed, alt, pathLength, tTOL, tPath, tWP, t0 = m_startTime, t1 = m_endTime;
1147 size_t w;
1148
1149 //------------------ MAGIC
1150 alt = altitudeMagic(task);
1151 //---------------------------
1152 pathLength = alt;
1153 previousWaypoint = path[0];
1154 for (w = 1; w < path.size(); w++)
1155 {
1156 pathLength += haversine(previousWaypoint, path.at(w));
1157 }
1158 pathLength += alt;
1159 tPath = t1 - t0;
1160 tTOL = tPath * alt / pathLength;
1161 v = pathLength / tPath;

1168 t0 = t0 + tTOL;
1169
1170 previousWaypoint = path[0];
1171 for (w = 1; w < path.size(); w++)
1172 {
1173 tWP = tPath * haversine(previousWaypoint, path.at(w)) / pathLength;
1174 cmd = fillCommand(CommandType::NAV_WAYPOINT, task, t0, t0 + tWP, { 0, 0, 0, 0, path.at(w).x(), path.at(w).y(), alt });
1175 m_cmdList.push_back(cmd);
1176 t0 = t0 + tWP;
1177
1178 previousWaypoint = path[w];
1179 }
1180 w--;
1181
1182 cmd = fillCommand(CommandType::NAV_LAND, task, t0, t1, { 0, 0, 0, path.at(w).x(), path.at(w).y(), alt });
1183 m_cmdList.push_back(cmd);
